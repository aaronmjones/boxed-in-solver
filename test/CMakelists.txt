
enable_testing()
find_package(GTest REQUIRED)

add_executable(
  encoded_path_test
  encoded_path_test.cc
)

target_include_directories(
  encoded_path_test PRIVATE
    ${CMAKE_SOURCE_DIR}/hdr
    ${Boost_INCLUDE_DIRS}
)

target_link_libraries(
  encoded_path_test
  GTest::GTest
  GTest::Main
)

gtest_discover_tests(encoded_path_test)
